CREATE TABLE Demographics
(ID INT NOT NULL AUTO_INCREMENT,
FName VARCHAR(200),
Age VARCHAR(200),
Race VARCHAR(200),
Gender VARCHAR(200),
Major VARCHAR(200),
YearofD VARCHAR(200),
Grade VARCHAR(200),
Experience VARCHAR(200),
Expertise VARCHAR(200),
FirstLanguage VARCHAR(200),
NofLanguages VARCHAR(200),
PRIMARY KEY (ID)
);
